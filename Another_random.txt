def find_missing_asserts(assert_list_file, target_file):
    # Read the list of assertions
    with open(assert_list_file, "r") as f:
        asserts = [line.strip() for line in f if line.strip()]

    # Read the entire target file as text
    with open(target_file, "r") as f:
        target_lines = f.readlines()

    missing = []

    for assertion in asserts:
        found = any(assertion in line for line in target_lines)
        if not found:
            missing.append(assertion)

    return missing


if __name__ == "__main__":
    list_file = "fpv_asrt_list.txt"
    target_file = "target.txt"

    missing_asserts = find_missing_asserts(list_file, target_file)

    if missing_asserts:
        print("Assertions NOT found in target file:")
        for a in missing_asserts:
            print(a)
